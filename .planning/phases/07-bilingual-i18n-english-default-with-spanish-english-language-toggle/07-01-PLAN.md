    2. Create `src/lib/i18n/context.tsx`:
       ```
       "use client";
       import { createContext, useState, useEffect, useCallback } from "react";
       import type { Locale } from "./types";
       import { DEFAULT_LOCALE, LOCALE_STORAGE_KEY } from "./types";
       import { translations } from "./translations";
       ```
       - `LocaleContext` = createContext with { locale, setLocale, t }
       - `LocaleProvider` component:
         - State: `locale` initialized to `DEFAULT_LOCALE` ('en')
         - useEffect on mount:
           a. Check `localStorage.getItem(LOCALE_STORAGE_KEY)` -- if 'en' or 'es', use it
           b. Else check `navigator.language` -- if starts with 'es', use 'es', else 'en'
           c. Set state
         - `setLocale(newLocale)` updates state AND writes to localStorage
         - `t(key: string): string` returns `translations[locale][key] ?? key`
         - Provides { locale, setLocale, t } through context

    3. Create `src/lib/i18n/use-locale.ts`:
       ```
       "use client";
       import { useContext } from "react";
       import { LocaleContext } from "./context";
       export function useLocale() {
         const ctx = useContext(LocaleContext);
         if (!ctx) throw new Error("useLocale must be used within LocaleProvider");
         return ctx;
       }
       ```

       - Re-export everything from types, translations, context, use-locale
